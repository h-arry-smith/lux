#+begin_src
  selection -> selector block
  selector -> "[" NUMBER "]"
  
  block -> "{" expression* "}"

  expression -> identifier value NEW_LINE
  identifier -> ALPHA+ ":" " "
  value -> NUMBER

  
  NUMBER -> NUMERIC ( "." NUMERIC+ )?
  NEW_LINE -> "\n"
#+end_src
